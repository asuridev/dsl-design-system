        +Uuid id
        +String name
        +String description
        +Uuid parentCategoryId

        +reactivate() void
    }

    class CategoryStatus {
        <<enumeration>>
        ACTIVE
        INACTIVE
    }

    class Product {
        +Uuid id
        +String name
        +String sku
        +Text description
        +Money price
        +Uuid categoryId
        +ProductStatus status
        +DateTime createdAt
        +DateTime updatedAt
        +create(name, sku, description, price, categoryId) Product
        +update(name, description, price, categoryId) void
        +activate() void
        +discontinue() void
        +addImage(url, altText, imageType, displayOrder) void
        +removeImage(imageId) void
    }

    class ProductStatus {
        <<enumeration>>
        DRAFT
        ACTIVE
        DISCONTINUED
    }

    class ProductImage {
        +Uuid id
        +Url url
        +String altText
        +ImageType imageType
        +Integer displayOrder
    }

    class ImageType {
        <<enumeration>>
        MAIN
        GALLERY
        THUMBNAIL
    }

    class Money {
        <<valueObject>>
        +Decimal amount
        +String currency
    }

    class ProductActivated {
        <<domainEvent>>
        +Uuid productId
        +String sku
        +String name
        +Uuid categoryId
    }

    class ProductDiscontinued {
        <<domainEvent>>
        +Uuid productId
        +String sku
    }

sequenceDiagram
    participant Admin
    participant CatalogAPI as Catalog API
    participant ProductAggregate as Product Aggregate
    participant Outbox
    participant Broker as Message Broker
    participant Inventory as Inventory BC

    Admin->>CatalogAPI: PATCH /products/{id}/activate
    CatalogAPI->>ProductAggregate: load Product by id
    CatalogAPI->>CatalogAPI: load Category by product.categoryId
    CatalogAPI->>CatalogAPI: validate CAT-RULE-001 (name not empty, price > 0, category ACTIVE)
    CatalogAPI->>ProductAggregate: activate()
    ProductAggregate-->>CatalogAPI: status = ACTIVE, event ProductActivated raised
    CatalogAPI->>Outbox: persist ProductActivated { productId, sku, name, categoryId }
    CatalogAPI-->>Admin: 204 No Content

    Outbox->>Broker: publish to catalog.product.activated
    Broker->>Inventory: deliver ProductActivated
    Inventory->>Inventory: create StockItem for productId

sequenceDiagram
    participant Admin
    participant CatalogAPI as Catalog API
    participant ProductAggregate as Product Aggregate
    participant Outbox
    participant Broker as Message Broker
    participant Inventory as Inventory BC

    Admin->>CatalogAPI: PATCH /products/{id}/discontinue
    CatalogAPI->>ProductAggregate: load Product by id
    CatalogAPI->>ProductAggregate: discontinue()
    ProductAggregate-->>CatalogAPI: status = DISCONTINUED, event ProductDiscontinued raised
    CatalogAPI->>Outbox: persist ProductDiscontinued { productId, sku }
    CatalogAPI-->>Admin: 204 No Content

    Outbox->>Broker: publish to catalog.product.discontinued
    Broker->>Inventory: deliver ProductDiscontinued
    Inventory->>Inventory: permanently close StockItem for productId
